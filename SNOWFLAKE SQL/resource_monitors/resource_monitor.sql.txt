-- =====================================================
-- PHASE 5 & 6 SAFE VERSION: RESOURCE MONITORS + MONITORING VIEWS
-- =====================================================

USE ROLE ACCOUNTADMIN;

-- ================================
-- 1️⃣ ACCOUNT-LEVEL RESOURCE MONITOR
-- ================================
CREATE OR REPLACE RESOURCE MONITOR HC_ACCOUNT_MONITOR
  WITH CREDIT_QUOTA = 100
  TRIGGERS
    ON 80 PERCENT DO NOTIFY
    ON 100 PERCENT DO SUSPEND;

-- ================================
-- 2️⃣ WAREHOUSE-LEVEL RESOURCE MONITORS
-- ================================
CREATE OR REPLACE RESOURCE MONITOR HC_ETL_WH_MONITOR
  WITH CREDIT_QUOTA = 50
  TRIGGERS
    ON 80 PERCENT DO NOTIFY
    ON 100 PERCENT DO SUSPEND;
ALTER WAREHOUSE HC_ETL_WH SET RESOURCE_MONITOR = HC_ETL_WH_MONITOR;

CREATE OR REPLACE RESOURCE MONITOR HC_TRANSFORM_WH_MONITOR
  WITH CREDIT_QUOTA = 70
  TRIGGERS
    ON 80 PERCENT DO NOTIFY
    ON 100 PERCENT DO SUSPEND;
ALTER WAREHOUSE HC_TRANSFORM_WH SET RESOURCE_MONITOR = HC_TRANSFORM_WH_MONITOR;

CREATE OR REPLACE RESOURCE MONITOR HC_ANALYTICS_WH_MONITOR
  WITH CREDIT_QUOTA = 30
  TRIGGERS
    ON 80 PERCENT DO NOTIFY
    ON 100 PERCENT DO SUSPEND;
ALTER WAREHOUSE HC_ANALYTICS_WH SET RESOURCE_MONITOR = HC_ANALYTICS_WH_MONITOR;

CREATE OR REPLACE RESOURCE MONITOR HC_AI_WH_MONITOR
  WITH CREDIT_QUOTA = 100
  TRIGGERS
    ON 80 PERCENT DO NOTIFY
    ON 100 PERCENT DO SUSPEND;
ALTER WAREHOUSE HC_AI_WH SET RESOURCE_MONITOR = HC_AI_WH_MONITOR;

USE DATABASE MONITORING_DB;

-- =====================================================
-- PHASE 6: MONITORING VIEWS (Safe Version)
-- =====================================================

USE ROLE ACCOUNTADMIN;

-- Create monitoring database and schema
CREATE OR REPLACE DATABASE MONITORING_DB;
USE DATABASE MONITORING_DB;

CREATE OR REPLACE SCHEMA MONITORING_SCHEMA;
USE SCHEMA MONITORING_SCHEMA;

-- 1️⃣ Query History
CREATE OR REPLACE VIEW QUERY_HISTORY_VIEW AS
SELECT * FROM SNOWFLAKE.ACCOUNT_USAGE.QUERY_HISTORY;

-- 2️⃣ Warehouse Load History
CREATE OR REPLACE VIEW WAREHOUSE_LOAD_HISTORY_VIEW AS
SELECT * FROM SNOWFLAKE.ACCOUNT_USAGE.WAREHOUSE_LOAD_HISTORY;

-- 3️⃣ Login History
CREATE OR REPLACE VIEW LOGIN_HISTORY_VIEW AS
SELECT * FROM SNOWFLAKE.ACCOUNT_USAGE.LOGIN_HISTORY;

-- 4️⃣ Failed Logins
CREATE OR REPLACE VIEW FAILED_LOGINS_VIEW AS
SELECT * FROM SNOWFLAKE.ACCOUNT_USAGE.LOGIN_HISTORY
WHERE IS_SUCCESS = 'FALSE';

-- 5️⃣ Blocked Queries
CREATE OR REPLACE VIEW BLOCKED_QUERIES_VIEW AS
SELECT * FROM SNOWFLAKE.ACCOUNT_USAGE.QUERY_HISTORY
WHERE ERROR_CODE IS NOT NULL;

-- 6️⃣ Long Running Queries (> 1 hour)
CREATE OR REPLACE VIEW LONG_RUNNING_QUERIES_VIEW AS
SELECT * FROM SNOWFLAKE.ACCOUNT_USAGE.QUERY_HISTORY
WHERE TOTAL_ELAPSED_TIME > 3600000;

-- Grant access to admin role
GRANT USAGE ON DATABASE MONITORING_DB TO ROLE HC_ACCOUNT_ADMIN;
GRANT USAGE ON SCHEMA MONITORING_SCHEMA TO ROLE HC_ACCOUNT_ADMIN;
GRANT SELECT ON ALL VIEWS IN SCHEMA MONITORING_SCHEMA TO ROLE HC_ACCOUNT_ADMIN;
