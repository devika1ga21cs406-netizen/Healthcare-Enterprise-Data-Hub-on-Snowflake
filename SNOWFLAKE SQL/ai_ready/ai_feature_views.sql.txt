USE ROLE SYSADMIN;

-- 1️⃣ Ensure AI_READY_DB and schema exist
CREATE DATABASE IF NOT EXISTS AI_READY_DB;
CREATE SCHEMA IF NOT EXISTS AI_READY_DB.AI_SCHEMA;
USE DATABASE AI_READY_DB;
USE SCHEMA AI_SCHEMA;

-- 2️⃣ Create a synthetic ICU dataset (10k patients)
CREATE OR REPLACE TABLE ICU_SYNTHETIC_RAW AS
SELECT
    SEQ4() AS PATIENT_ID,
    'Patient_' || LPAD(SEQ4(),5,'0') AS NAME,
    UNIFORM(18,90,RANDOM()) AS AGE,
    CASE WHEN UNIFORM(0,1,RANDOM()) < 0.5 THEN 'M' ELSE 'F' END AS GENDER,
    DATEADD(DAY, -UNIFORM(0,365,RANDOM()), CURRENT_DATE) AS ADMISSION_DATE,
    CASE
        WHEN UNIFORM(0,1,RANDOM()) < 0.3 THEN 'Cardiology'
        WHEN UNIFORM(0,1,RANDOM()) < 0.6 THEN 'Neurology'
        ELSE 'General'
    END AS DIAGNOSIS,
    UNIFORM(50,150,RANDOM()) AS HEART_RATE,
    UNIFORM(80,100,RANDOM()) AS OXYGEN_LEVEL,
    UNIFORM(1000,50000,RANDOM()) AS BILL_AMOUNT,
    CASE
        WHEN UNIFORM(0,1,RANDOM()) < 0.3 THEN 'Patient shows fever and cough'
        WHEN UNIFORM(0,1,RANDOM()) < 0.6 THEN 'Patient has headache and dizziness'
        ELSE 'Patient came for general checkup'
    END AS PATIENT_NOTES
FROM TABLE(GENERATOR(ROWCOUNT => 10000));

-- 3️⃣ Create ICU_FEATURES table from synthetic data
CREATE OR REPLACE TABLE ICU_FEATURES AS
SELECT
    PATIENT_ID,
    NAME,
    AGE,
    GENDER,
    ADMISSION_DATE,
    DIAGNOSIS,
    HEART_RATE AS AVG_HEART_RATE,
    OXYGEN_LEVEL AS AVG_OXYGEN_LEVEL,
    DATEDIFF(HOUR, ADMISSION_DATE, CURRENT_TIMESTAMP) AS LENGTH_OF_STAY_HOURS,
    BILL_AMOUNT AS BILLING_RISK_SCORE,
    CASE WHEN OXYGEN_LEVEL < 90 THEN 1 ELSE 0 END AS RISK_LOW_OXYGEN,
    CASE WHEN AGE > 60 THEN 1 ELSE 0 END AS RISK_AGE,
    PATIENT_NOTES
FROM ICU_SYNTHETIC_RAW;

-- 4️⃣ Create synthetic embeddings (placeholder)
CREATE OR REPLACE TABLE PATIENT_EMBEDDINGS AS
SELECT
    NAME,
    ARRAY_CONSTRUCT(
        UNIFORM(0,1,RANDOM()),
        UNIFORM(0,1,RANDOM()),
        UNIFORM(0,1,RANDOM()),
        UNIFORM(0,1,RANDOM())
    ) AS PATIENT_EMBEDDING
FROM ICU_FEATURES;

-- 5️⃣ Semantic view combining features + embeddings
CREATE OR REPLACE VIEW ICU_FEATURES_VIEW AS
SELECT
    f.*,
    e.PATIENT_EMBEDDING
FROM ICU_FEATURES f
LEFT JOIN PATIENT_EMBEDDINGS e
ON f.NAME = e.NAME;

-- 6️⃣ Verification queries
SELECT * FROM ICU_FEATURES LIMIT 5;
SELECT * FROM PATIENT_EMBEDDINGS LIMIT 5;
SELECT * FROM ICU_FEATURES_VIEW LIMIT 5;
