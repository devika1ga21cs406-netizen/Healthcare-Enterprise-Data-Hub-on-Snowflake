USE ROLE ACCOUNTADMIN;

-- Set database & schema context
USE DATABASE MONITORING_DB;
USE SCHEMA MONITORING_SCHEMA;

-- 1️⃣ High Credit Usage Alert
CREATE OR REPLACE ALERT MONITORING_DB.MONITORING_SCHEMA.HC_CREDIT_ALERT
WAREHOUSE = HC_ANALYTICS_WH
SCHEDULE = '1 MINUTE'
IF (
    EXISTS(
        SELECT 1
        FROM SNOWFLAKE.ACCOUNT_USAGE.CREDIT_USAGE
        WHERE USAGE_DATE = CURRENT_DATE()
          AND CREDITS_USED > 80
    )
)
THEN CALL SYSTEM$SEND_EMAIL(
    'devika1ga21cs406@gmail.com',
    'High Credit Usage Alert',
    'Credit usage exceeded threshold today.'
);

-- 2️⃣ Warehouse Queue > 5 min
CREATE OR REPLACE ALERT MONITORING_DB.MONITORING_SCHEMA.HC_WH_QUEUE_ALERT
WAREHOUSE = HC_ANALYTICS_WH
SCHEDULE = '1 MINUTE'
IF (
    EXISTS(
        SELECT 1
        FROM SNOWFLAKE.ACCOUNT_USAGE.WAREHOUSE_LOAD_HISTORY
        WHERE WAREHOUSE_NAME = 'HC_ANALYTICS_WH'
          AND AVG_QUEUE_LOAD > 5*60*1000
    )
)
THEN CALL SYSTEM$SEND_EMAIL(
    'devika1ga21cs406@gmail.com',
    'Warehouse Queue Alert',
    'Warehouse queue time exceeded 5 minutes.'
);

-- 3️⃣ Failed Logins > 10 in last 1 hour
CREATE OR REPLACE ALERT MONITORING_DB.MONITORING_SCHEMA.HC_FAILED_LOGIN_ALERT
WAREHOUSE = HC_ANALYTICS_WH
SCHEDULE = '5 MINUTE'
IF (
    EXISTS(
        SELECT 1
        FROM SNOWFLAKE.ACCOUNT_USAGE.LOGIN_HISTORY
        WHERE IS_SUCCESS = 'FALSE'
          AND EVENT_TIMESTAMP >= DATEADD(HOUR, -1, CURRENT_TIMESTAMP())
        HAVING COUNT(*) > 10
    )
)
THEN CALL SYSTEM$SEND_EMAIL(
    'devika1ga21cs406@gmail.com',
    'Failed Login Alert',
    'More than 10 failed login attempts in the past hour.'
);

-- 4️⃣ Data Load Failure (COPY INTO failed)
CREATE OR REPLACE ALERT MONITORING_DB.MONITORING_SCHEMA.HC_DATA_LOAD_ALERT
WAREHOUSE = HC_ETL_WH
SCHEDULE = '5 MINUTE'
IF (
    EXISTS(
        SELECT 1
        FROM MONITORING_DB.MONITORING_SCHEMA.BLOCKED_QUERIES_VIEW
        WHERE QUERY_TEXT ILIKE '%COPY INTO%'
    )
)
THEN CALL SYSTEM$SEND_EMAIL(
    'devika1ga21cs406@gmail.com',
    'Data Load Failure Alert',
    'A data load command failed in the past 5 minutes.'
);

-- 5️⃣ Long Running Query > 30 min
CREATE OR REPLACE ALERT MONITORING_DB.MONITORING_SCHEMA.HC_LONG_QUERY_ALERT
WAREHOUSE = HC_ANALYTICS_WH

    'devika1ga21cs406@gmail.com',

);

-- 6️⃣ ICU Event: Heart Rate > 120
